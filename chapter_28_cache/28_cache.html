28.1 为什么 Agent 场景特别需要缓存？

 

Agent 的 LLM 调用特征：

  • 重复率高：用户问相似的问题（客服/教育场景 30-50% 重复）
  • 代价高：每次调用 = 金钱 + 时间
  • 步骤多：一个 Agent 任务可能调 5-10 次 LLM

    •  

    缓存收益估算：

    假设：日请求 10000 次，30% 可缓存，每次 $0.01

    日节省：10000 × 0.3 × $0.01 = $30/天

    年节省：$30 × 365 = $10,950/年

     

    加上延迟改善：缓存命中 P99 < 10ms vs LLM 调用 P99 > 2s)

     

     

    28.2 三级缓存架构

     

      ┌─────────────────────────────────────────────────┐
  │                 L1: Exact Match Cache              │
  │  ─────────────────────────────────               │
  │  查询: "今天天气" → 缓存中查 "今天天气"             │
  │  命中条件: 完全相同的文本                           │
  │  命中率: 5-10%                                    │
  │  延迟: <1ms                                       │
  ├─────────────────────────────────────────────────┤
  │                 L2: Semantic Cache                 │
  │  ─────────────────────────────────               │
  │  查询: "今天天气" → 缓存中查相似查询 → 命中 "今天天气怎么样"│
  │  命中条件: 余弦相似度 > 0.95                        │
  │  命中率: 20-40%                                    │
  │  延迟: ~10ms                                       │
  ├─────────────────────────────────────────────────┤
  │                 L3: LLM-as-Cache                   │
  │  ─────────────────────────────────               │
  │  查询: 无法在前2级命中 → 调用廉价小模型               │
  │  如果小模型回答合适 → 缓存结果                       │
  │  命中率: 10-20% (加上前两级：总 40-70%)              │
  │  延迟: ~500ms                                      │
  └─────────────────────────────────────────────────┘

class ExactMatchCache:
    class=class="s">"s">class="s">"""L1 缓存 —— 精确匹配。

    最简单的缓存：完全相同的输入 → 直接返回缓存结果。
    class="s">"""

    def __init__(self, max_size: int = 1000):
        self._store = OrderedDict()
        self.max_size = max_size
        self.hits = 0
        self.misses = 0

    def get(self, query: str) -> Optional[str]:
        key = hashlib.md5(query.encode()).hexdigest()
        if key in self._store:
            self._store.move_to_end(key)  class=class="s">"h"># LRU 更新
            self.hits += 1
            return self._store[key]
        self.misses += 1
        return None

    def set(self, query: str, response: str):
        key = hashlib.md5(query.encode()).hexdigest()
        if len(self._store) >= self.max_size:
            self._store.popitem(last=False)  class=class="s">"h"># 淘汰最久未用
        self._store[key] = response

    @property
    def hit_rate(self) -> float:
        total = self.hits + self.misses
        return self.hits / total if total > 0 else 0


class SemanticCache:
    class=class="s">"s">class="s">"""L2 缓存 —— 语义相似度匹配。

    不要求完全相同的查询，语义相似即可命中。
    class="s">"""

    def __init__(self, similarity_threshold: float = 0.92,
                 max_size: int = 2000):
        self.threshold = similarity_threshold
        self.max_size = max_size
        self.queries = []      class=class="s">"h"># [(embedding, query, response)]
        self.hits = 0
        self.misses = 0

    def _embed(self, text: str, dim: int = 128) -> np.ndarray:
        class=class="s">"s">class="s">""class="s">"简化的向量化（实际使用 OpenAI Embeddings API）。"class="s">""
        vec = np.zeros(dim)
        for i, ch in enumerate(text):
            vec[hash(ch) % dim] += 1
        norm = np.linalg.norm(vec)
        return vec / norm if norm > 0 else vec

    def get(self, query: str) -> Optional[str]:
        q_vec = self._embed(query)
        for emb, cached_query, response in self.queries:
            sim = float(np.dot(q_vec, emb))
            if sim >= self.threshold:
                self.hits += 1
                return response
        self.misses += 1
        return None

    def set(self, query: str, response: str):
        emb = self._embed(query)
        self.queries.append((emb, query, response))
        if len(self.queries) > self.max_size:
            self.queries = self.queries[-self.max_size:]

    @property
    def hit_rate(self) -> float:
        total = self.hits + self.misses
        return self.hits / total if total > 0 else 0


class ThreeLevelCache:
    class=class="s">"s">class="s">""class="s">"三级缓存 —— 层层降级，最大化命中率。"class="s">""

    def __init__(self):
        self.l1 = ExactMatchCache(max_size=500)
        self.l2 = SemanticCache(similarity_threshold=0.90)
        self.stats = {class="s">"l1_hits": 0, class="s">"l2_hits": 0,
                      class="s">"l3_hits": 0, class="s">"misses": 0,
                      class="s">"cost_saved": 0.0}
        self.COST_PER_LLM_CALL = 0.01  class=class="s">"h"># 每次 LLM 调用成本

    def query(self, query: str,
              llm_func: callable = None) -> tuple[str, int, bool]:
        class=class="s">"s">class="s">"""三级缓存查询。

        Returns:
            (response, cache_level, was_cached)
            cache_level: 1=L1, 2=L2, 3=L3, 0=miss
        class="s">"""
        class=class="s">"h"># L1: Exact Match
        result = self.l1.get(query)
        if result:
            self.stats[class="s">"l1_hits"] += 1
            self.stats[class="s">"cost_saved"] += self.COST_PER_LLM_CALL
            return result, 1, True

        class=class="s">"h"># L2: Semantic
        result = self.l2.get(query)
        if result:
            self.stats[class="s">"l2_hits"] += 1
            self.stats[class="s">"cost_saved"] += self.COST_PER_LLM_CALL
            return result, 2, True

        class=class="s">"h"># L3: LLM
        if llm_func:
            response = llm_func(query)
        else:
            response = fclass="s">"[LLM 响应] 关于「{query[:30]}...」的回答"
        self.stats[class="s">"l3_hits"] += 1

        class=class="s">"h"># 写入缓存
        self.l1.set(query, response)
        self.l2.set(query, response)

        return response, 3, False

    def report(self) -> dict:
        total = sum([self.stats[class="s">"l1_hits"], self.stats[class="s">"l2_hits"],
                     self.stats[class="s">"l3_hits"], self.stats[class="s">"misses"]])
        total = max(total, 1)
        return {
            class="s">"total_queries": total,
            class="s">"l1_hit_rate": fclass="s">"{self.stats[class='s'>'l1_hits'] / total:.1%}",
            class="s">"l2_hit_rate": fclass="s">"{self.stats[class='s'>'l2_hits'] / total:.1%}",
            class="s">"l3_rate": fclass="s">"{self.stats[class='s'>'l3_hits'] / total:.1%}",
            class="s">"total_cache_hit": (fclass="s">"{(self.stats[class='s'>'l1_hits'] + self.stats[class='s'>'l2_hits']) / total:.1%}"),
            class="s">"cost_saved": fclass="s">"${self.stats[class='s'>'cost_saved']:.3f}",
            class="s">"l1_size": len(self.l1._store),
            class="s">"l2_size": len(self.l2.queries),
        }

class TokenBudget:
    class=class="s">"s">class="s">""class="s">"Token 预算管理器 —— 类比流量套餐管理。"class="s">""

    def __init__(self, daily_limit: int = 1_000_000):
        self.daily_limit = daily_limit
        self.used_today = 0
        self.warning_threshold = 0.8
        self.critical_threshold = 0.95
        self.cost_per_1k = 0.002  class=class="s">"h"># $/1K tokens
        self.history = []

    def check(self, required_tokens: int) -> dict:
        class=class="s">"s">class="s">"""检查预算是否充足。

        Returns:
            预算状态。
        class="s">"""
        projected = self.used_today + required_tokens
        pct = projected / self.daily_limit

        if pct >= 1.0:
            return {class="s">"allowed": False, class="s">"status": class="s">"exceeded",
                    class="s">"message": fclass="s">"今日预算已用完 ({self.daily_limit:,} tokens)"}

        if pct >= self.critical_threshold:
            return {class="s">"allowed": True, class="s">"status": class="s">"critical",
                    class="s">"message": fclass="s">"接近预算上限 ({pct:.0%})，建议用小模型"}

        if pct >= self.warning_threshold:
            return {class="s">"allowed": True, class="s">"status": class="s">"warning",
                    class="s">"message": fclass="s">"预算已达 {pct:.0%}"}

        return {class="s">"allowed": True, class="s">"status": class="s">"ok"}

    def consume(self, tokens: int):
        class=class="s">"s">class="s">""class="s">"消耗预算。"class="s">""
        self.used_today += tokens
        self.history.append({
            class="s">"tokens": tokens,
            class="s">"remaining": self.daily_limit - self.used_today,
            class="s">"pct": self.used_today / self.daily_limit,
        })

    def daily_report(self) -> dict:
        class=class="s">"s">class="s">""class="s">"每日预算报告。"class="s">""
        pct = self.used_today / self.daily_limit
        cost = self.used_today / 1000 * self.cost_per_1k
        return {
            class="s">"daily_limit": fclass="s">"{self.daily_limit:,}",
            class="s">"used": fclass="s">"{self.used_today:,}",
            class="s">"remaining": fclass="s">"{self.daily_limit - self.used_today:,}",
            class="s">"usage_pct": fclass="s">"{pct:.1%}",
            class="s">"est_cost": fclass="s">"${cost:.2f}",
            class="s">"status": class="s">"🟢" if pct < 0.8 else (class="s">"🟡" if pct < 0.95 else class="s">"🔴"),
        }

28.4 缓存的进阶策略
━━━━━━━━━━━━━━━━━━━

1. 分层 TTL
   L1 精确缓存：永久（直到 LRU 淘汰）
   L2 语义缓存：24小时 TTL（信息会过时）
   L3 LLM 缓存：按查询类型设 TTL（新闻类 1h / 知识类 7天）

2. 缓存预热
   上线前用常见问题预填充缓存

3. 缓存失效
   知识库更新 → 相关缓存全部失效

4. 监控指标
   命中率 / 节省成本 / 缓存大小 / L1/L2/L3 分流比例
