def call_llm(prompt: str) -> str:
    class=class="s">"s">class="s">"""最基础的 LLM 调用 —— 发送提示词，获取回复。

    这是 Agent 的最底层操作：让 LLM「思考」。
    后续所有 Agent 逻辑都建立在这个基础之上。

    Args:
        prompt: 用户输入的提示词文本。

    Returns:
        LLM 生成的回复文本。
    class="s">"""
    model = os.getenv(class="s">"LLM_MODEL", class="s">"gpt-4o-mini")
    response = client.chat.completions.create(
        model=model,
        messages=[{class="s">"role": class="s">"user", class="s">"content": prompt}],
        temperature=0.7,
    )
    return response.choices[0].message.content

Function Calling 是 Agent 的核心机制。LLM 不只是生成文本，
它还能「决定」调用哪个函数、传什么参数。

流程：
  用户：「北京今天天气怎么样？」
  →

  LLM 输出: {class="s">"name": class="s">"get_weather", class="s">"arguments": {class="s">"city": class="s">"北京"}}
  →
  我们的代码执行这个函数，拿到真实数据
  →
  把结果返回给 LLM，LLM 组织成自然语言回答用户

类比：LLM 像一位「盲人指挥家」，他能决定需要什么信息，
但必须由我们（代码）去执行实际的工具调用。

def get_weather(city: str) -> str:

    Args:

        class="s">"北京": class="s">"晴天，25°C，湿度 40%，北风 3 级",

def calculate(expression: str) -> str:
    class=class="s">"s">class="s">"""安全的数学计算工具。

    Args:
        expression: 数学表达式字符串。

    Returns:
        计算结果字符串。
    class="s">"""
    try:
        allowed_chars = set(class="s">"0123456789+-*/().%^ ")
        if not all(c in allowed_chars for c in expression):
            return class="s">"错误：表达式包含不允许的字符"
        result = eval(expression)
        return fclass="s">"计算结果：{result}"
    except Exception as e:
        return fclass="s">"计算错误：{str(e)}"

1.3 手写 Agent 循环 —— 理解底层原理
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

这是本章最核心的部分！

每个 Agent 的运行本质上是一个循环：
  1. 把用户消息发给 LLM
  2. LLM 判断：直接回答？还是需要调用工具？
  3. 如果需要调用工具 → 执行工具 → 把结果发给 LLM，回到步骤 1
  4. 如果可以回答 → 输出答案，结束

这个循环被称为 ReAct 循环（Reasoning + Acting），
是 LangChain / AutoGPT / crewAI 等所有框架的底层逻辑。
class="s">"""


def run_agent(user_message: str, max_iterations: int = 5):
    class=class="s">"s">class="s">"""运行一个最简单的 ReAct Agent。

    这是不加任何框架的「裸写」Agent，帮助理解底层原理。
    任何 Agent 框架（LangChain/crewAI/AutoGen）本质上
    都是这个循环的封装 + 增强。

    Args:
        user_message: 用户输入的消息。
        max_iterations: 最大工具调用轮数，防止无限循环。

    Returns:
        Agent 的最终回答。
    class="s">"""
    class=class="s">"h"># 对话历史 —— Agent 的「短期记忆」
    messages = [
        {class="s">"role": class="s">"system", class="s">"content": class="s">"你是一个有用的 AI 助手。当需要获取最新信息或执行计算时，请使用提供的工具。"},
        {class="s">"role": class="s">"user", class="s">"content": user_message},
    ]

    print(fclass="s">"\n{class='s'>'='*60}")
    print(fclass="s">"用户: {user_message}")
    print(fclass="s">"{class='s'>'='*60}")

    for iteration in range(max_iterations):
        class=class="s">"h"># ===== 第 1 步：调用 LLM（Think 阶段）=====
        model = os.getenv(class="s">"LLM_MODEL", class="s">"gpt-4o-mini")
        response = client.chat.completions.create(
            model=model,
            messages=messages,
            tools=TOOLS,
            tool_choice=class="s">"auto",
        )

        assistant_message = response.choices[0].message

        class=class="s">"h"># ===== 第 2 步：检查 LLM 是否需要调用工具 =====
        if assistant_message.tool_calls is None:
            class=class="s">"h"># LLM 决定直接回答（不需要工具）→ 循环结束
            final_answer = assistant_message.content
            print(fclass="s">"\n🎯 Agent 最终回答:\n{final_answer}")
            return final_answer

        class=class="s">"h"># ===== 第 3 步：执行工具调用（Act 阶段）=====
        class=class="s">"h"># 将 LLM 的回复（含 tool_calls）加入对话历史
        messages.append(assistant_message)

        for tool_call in assistant_message.tool_calls:
            func_name = tool_call.function.name
            func_args = json.loads(tool_call.function.arguments)

            print(fclass="s">"\n🔧 第 {iteration + 1} 轮 - 调用工具: {func_name}({func_args})")

            class=class="s">"h"># 执行实际的工具函数
            tool_function = TOOL_MAP.get(func_name)
            if tool_function is None:
                result = fclass="s">"错误：未找到工具 {func_name}"
            else:
                result = tool_function(**func_args)

            print(fclass="s">"📊 工具返回: {result}")

            class=class="s">"h"># 将工具执行结果加入对话历史（Observe 阶段）
            messages.append({
                class="s">"role": class="s">"tool",
                class="s">"tool_call_id": tool_call.id,
                class="s">"content": result,
            })

    class=class="s">"h"># 达到最大迭代次数，强制 LLM 给出最终回答
    print(class="s">"\n⚠️ 达到最大迭代次数，要求 LLM 给出最终回答...")
    messages.append({
        class="s">"role": class="s">"user",
        class="s">"content": class="s">"请基于已有的工具调用结果，给出最终回答。"
    })
    model = os.getenv(class="s">"LLM_MODEL", class="s">"gpt-4o-mini")
    final_response = client.chat.completions.create(
        model=model,
        messages=messages,
    )
    final_answer = final_response.choices[0].message.content
    print(fclass="s">"\n🎯 Agent 最终回答:\n{final_answer}")
    return final_answer
